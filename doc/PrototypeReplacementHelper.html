def tooltip(name = 'Help', opts = {}, &block)
  button = content_tag(:span, name, class: 'btn btn-info popover-trigger', 'data-content': capture(opts, &block),
  'data-toggle': 'popover', 'data-title': opts.fetch(:title, 'About this'))
  concat button
end
