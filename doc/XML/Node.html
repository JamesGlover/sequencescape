def register_default_namespace(name)
  self.namespace.each do |n|
    if n.to_s == nil
      register_namespace("#{name}:#{n.href}")
      return
    end
  end
  raise "No default namespace found"
end

def search(xpath)
  results = self.find(xpath).to_a
  if block_given?
    results.each do |result|
      yield result
    end
  end
  return results
end
