def ended_billable_lanes(ended)
  events = []
  self.samples.each do |sample|
    if sample.ended.casecmp(ended).zero?
      events << sample.billable_events
    end
  end
  events = events.flatten
end

  a = []
  events.each do |event|

    end
  end
  a.size
end
