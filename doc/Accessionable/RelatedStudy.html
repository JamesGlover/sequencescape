def build(xml)
  return if db_label.blank?
  xml.RELATED_STUDY {
    xml.RELATED_LINK {
      xml.DB db_label
      xml.ID @study.ebi_accession_number
  }
    xml.IS_PRIMARY @primary
  }
end
