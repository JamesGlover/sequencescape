def close
  @response.close if @response.respond_to?(:close) # Pass us on
  stop = Time.now
  @output.syswrite "Request: #{@env['REQUEST_METHOD']}%#{@env['REQUEST_URI']}%#{@env["rack.input"].string}%#{stop - @start}\n"
