def initialize(method = nil, &block)
  if method.present?
    line = __LINE__ + 1
    singleton_class.class_eval("
      def execute(object)
        object.#{method}
      end
    ", __FILE__, line)
  elsif block_given?
    singleton_class.send(:define_method, :execute, &block)
  else
    raise StandardError, 'Either method name or block is required for guards'
  end
end
