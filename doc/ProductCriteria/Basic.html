def sample_gender
  markers = @well_or_metric.samples.map { |s| s.sample_metadata.gender && s.sample_metadata.gender.downcase.strip }.uniq
  return nil if markers.count > 1
  GENDER_MARKER_MAPS[markers.first]
end
